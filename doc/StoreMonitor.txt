StoreMonitor
	table structure
		users 
			id 序列号 serial 
			userNumber 账户 varchar(50) primary key
			userName 用户名 char(50)
			password 密码 varchar(50)
			level 人员等级 int 1：防盗员  2：防盗经理  3：区域经理
			phone 电话 varchar(11)
			email 邮箱 varchar(50)
		shops
			id 序列号 serial
			shopId 商店名 varchar(50)  primary key
			monitorMan 防盗员 varchar(50)---users
			manager 防盗经理 varchar(50)---users
			districtManager 区域经理varchar(50)---users
		lists (视频记录）
			id 序列号 serial
			transId 交易单号 varchar(50) primary key
			regId 银台编号 varchar(50)
			cashierId 收银员编号 varchar(50)
			tsStart 视频开始时间 int
			tsEnd 视频结束时间 int
			scriptVer 版本号 varchar(50)
			vedioUrl 视频URL varchar(200)
			priority 优先级 int
			createAt 创建时间 timestamp not null
			updateAt 更新时间 timestamp not null
			state 事件状态 varchar(50) ---events.type
			result 审核结果 varchar(50)
			shopId 商店名字 varchar(50) ---shops
		styles (事件渲染类型）
			id 序列号 serial
			style 标注类型varchar(50) primary key
			fontType字体类型 varchar(50)
			fontSize 字体大小 int
			fontRGB 颜色 varchar(50)
			iconUrl 图标URL varchar(200)
			iconW 图标宽度 int
			iconH 图标高度 int
			transId交易单号 varchar(50) ---lists
		events(可疑事件）
			id 序列号 serial primary key
			eventStart 叠加开始时间 varchar(50)
			eventEnd 叠加结束时间 varchar(50)
			text 文字内容 varchar(50)
			centerX 叠加横坐标 int
			centerY 叠加纵坐标 int
			hide 隐藏 boolean true:隐藏 false:不隐藏
			type 视频类型 varchar(50) 'NoScan': 漏扫 'CartLeft':遗漏 'NorMal':正常
			transId 交易单号 varchar(50) --- lists
			style标注类型 varchar(50) --- styles
