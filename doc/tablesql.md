### users

```postgre

CREATE TABLE public.users
(
    id varchar(50) primary key,
    username varchar(50),
    password varchar(50) not null,
    phone varchar(11),
    email varchar(50),
    authorityId varchar(50)
)
WITH (
    OIDS=FALSE
);
ALTER TABLE public.users
OWNER TO company;

  ````
